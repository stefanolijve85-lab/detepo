<script>
// ── Bar chart data ──
const hourData = [2,1,4,1,0,1,5,12,18,22,28,31,35,38,42,36,30,24,18,14,10,7,4,2];
const peakHour = 14;

function buildChart() {
  const container = document.getElementById('barChart');
  if (!container) return;
  const max = Math.max(...hourData);
  const chartH = 68;
  container.innerHTML = '';
  hourData.slice(8, 20).forEach((v, i) => {
    const hour = i + 8;
    const wrap = document.createElement('div');
    wrap.className = 'bar-wrap';
    const bar = document.createElement('div');
    bar.className = 'bar-in';
    const h = Math.max((v / max) * chartH, 2);
    bar.style.height = h + 'px';
    bar.style.width = '100%';
    bar.style.opacity = hour === peakHour ? '1' : '0.45';
    if (hour === peakHour) {
      bar.style.background = 'var(--amber)';
      bar.style.boxShadow = '0 0 8px rgba(255,176,32,0.4)';
    }
    const lbl = document.createElement('div');
    lbl.className = 'bar-lbl';
    lbl.textContent = hour % 3 === 0 ? hour + '' : '';
    wrap.appendChild(bar);
    wrap.appendChild(lbl);
    container.appendChild(wrap);
  });
}

// ── Screen switching ──
const screens = ['home', 'live', 'insights', 'devices', 'alerts'];

function switchScreen(name, btn) {
  // Update tab buttons
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  btn.classList.add('active');

  // Show/hide phones
  screens.forEach(s => {
    const el = document.getElementById('phone-' + s);
    if (el) el.style.display = s === name ? 'flex' : 'none';
  });

  if (name === 'insights') buildChart();
}
